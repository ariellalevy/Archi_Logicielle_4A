    public static  Position give_order(char[] command, Position position, List<Position> setObstaclesPositions, Plateau map){ 
        for(int  i =0; i< command.length; i++){
            String cmd ="";
            cmd +=command[i];
            position = move(cmd, position, setObstaclesPositions, map);
        }
        return position;
    }
